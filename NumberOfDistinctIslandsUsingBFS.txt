#include<bits/stdc++.h>
using namespace std;
typedef pair<int, int> P;

void getLand(vector<vector<int>> &grid, vector<vector<int>> &visited, set<vector<P>> &distinctLand, vector<P> &land, int i, int j) {
    int n = grid.size();
    int m = grid[0].size();
    int delRow[] = {0, 1, 0, -1};
    int delCol[] = {1, 0, -1, 0};
    queue<pair<int, int>> q;
    q.push({i, j});
    while(!q.empty()) {
        int frontRow = q.front().first;
        int frontCol = q.front().second;
        q.pop();
        for(int i = 0; i < 4; i++) {
            int newRow = frontRow + delRow[i];
            int newCol = frontCol + delCol[i];
            if(newRow >= 0 && newRow < n && newCol >= 0 && newCol < m && grid[newRow][newCol] == 1 && !visited[newRow][newCol]) {
                visited[newRow][newCol] = 1;
                q.push({newRow, newCol});
                land.push_back({newRow - frontRow, newCol - frontCol});
            }
        }
    }
}

int main() {
    
    int n, m;
    cout<<"Enter number of rows :- ";
    cin>>n;
    cout<<"Enter number of columns :- ";
    cin>>m;
    vector<vector<int>> grid(n, vector<int>(m));
    
    for(int i = 0; i < n; i++) {
        cout<<"Enter "<<i<<"th row :- ";
        for(int j = 0; j < m; j++) {
            cin>>grid[i][j];
        }
    }
    cout<<"2D matrix is :- "<<endl;
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cout<<grid[i][j]<<" ";
        }
        cout<<endl;
    }
    
    set<vector<P>> distinctLand;
    vector<vector<int>> visited(n, vector<int>(m, 0));
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            if(grid[i][j] == 1 && !visited[i][j]) {
                visited[i][j] = 1;
                vector<P> land;
                getLand(grid, visited, distinctLand, land, i, j);
                distinctLand.insert(land);
            }
        }
    }
    
    cout<<"Number of distinct land is :- "<<distinctLand.size();
    
    return 0;
}
